Core Algorithms & Models

Advanced Computational Methods

Hopkins Imaging

Partially coherent imaging using Hopkins' method with transmission cross coefficients (TCCs) for accurate aerial image calculation.

Abbe-Rayleigh Theory

Fourier optics-based diffraction modeling with full vector calculations for high-NA systems.

Kirchhoff Approximation

Mask diffraction modeling with thin-mask and thick-mask approximations for EUV and DUV lithography.

SOCS Decomposition

Sum of Coherent Systems decomposition for fast aerial image calculation with GPU acceleration.

Neural OPC

Deep learning-based OPC using convolutional neural networks trained on millions of pattern-correction pairs.

Zernike Polynomials

Aberration modeling using Zernike polynomial expansion up to 37 terms for lens characterization.

Implementation Example

Python Code for Complete Photolithography Simulation

# Advanced Photolithography Simulation Pipeline
import numpy as np
from scipy import ndimage, optimize
from sklearn.neural_network import MLPRegressor
import matplotlib.pyplot as plt

class PhotolithographySimulator:
    def __init__(self, wavelength=193e-9, NA=1.35, sigma=0.85):
        self.wavelength = wavelength
        self.NA = NA
        self.sigma = sigma
        self.k1 = 0.61  # Rayleigh criterion
        
    def calculate_resolution(self):
        """Calculate theoretical resolution limit"""
        return self.k1 * self.wavelength / self.NA
    
    def generate_mask_pattern(self, size=(512, 512), pitch=65e-9, duty_cycle=0.5):
        """Generate advanced mask pattern with PSM and SRAF"""
        x, y = np.meshgrid(np.linspace(0, size[0]*5e-9, size[0]),
                          np.linspace(0, size[1]*5e-9, size[1]))
        
        # Main pattern
        mask = ((x % pitch) < (pitch * duty_cycle)).astype(float)
        
        # Add phase shift mask (alternating phase)
        phase_mask = np.zeros_like(mask, dtype=complex)
        phase_mask[mask == 1] = 1.0
        phase_mask[::2, mask[::2] == 1] = np.exp(1j * np.pi)  # 180° phase shift
        
        # Add sub-resolution assist features
        sraf_width = pitch * 0.1
        for i in range(1, size[0], int(pitch/5e-9)):
            if i % 2 == 0:
                mask[i:i+int(sraf_width/5e-9), :] = 0.3
        
        return mask, phase_mask
    
    def apply_opc(self, mask, target, iterations=5, learning_rate=0.1):
        """Model-based OPC with machine learning enhancement"""
        corrected = mask.copy()
        
        # Neural network for EPE prediction
        model = MLPRegressor(hidden_layer_sizes=(100, 50), max_iter=100)
        
        for iter in range(iterations):
            # Simulate aerial image
            aerial = self.calculate_aerial_image(corrected)
            
            # Calculate edge placement error
            epe = self.calculate_epe(aerial, target)
            
            # Apply corrections
            gradient = ndimage.sobel(epe)
            corrected -= learning_rate * gradient
            corrected = np.clip(corrected, 0, 1)
            
            # ML-based refinement
            if iter > 2:
                features = self.extract_features(corrected)
                corrections = model.predict(features.reshape(1, -1))
                corrected += corrections.reshape(corrected.shape) * 0.1
        
        return corrected
    
    def calculate_aerial_image(self, mask):
        """Calculate aerial image using Hopkins method"""
        # Fourier transform of mask
        mask_spectrum = np.fft.fft2(mask)
        
        # Pupil function
        pupil = self.generate_pupil_function(mask.shape)
        
        # Partially coherent imaging
        aerial = np.zeros_like(mask, dtype=float)
        
        # Source points for partial coherence
        n_source = 9
        sigma_x = np.linspace(-self.sigma, self.sigma, n_source)
        sigma_y = np.linspace(-self.sigma, self.sigma, n_source)
        
        for sx in sigma_x:
            for sy in sigma_y:
                # Shifted pupil
                shifted_pupil = self.shift_pupil(pupil, sx, sy)
                
                # Coherent image for this source point
                coherent = np.fft.ifft2(mask_spectrum * shifted_pupil)
                aerial += np.abs(coherent)**2
        
        aerial /= (n_source * n_source)
        return aerial
    
    def detect_defects(self, image, reference, threshold=0.1):
        """Multi-algorithm defect detection"""
        # Die-to-die comparison
        diff = np.abs(image - reference)
        
        # Threshold detection
        defects = diff > threshold
        
        # Morphological operations for noise reduction
        defects = ndimage.binary_opening(defects, iterations=2)
        defects = ndimage.binary_closing(defects, iterations=2)
        
        # Label and analyze defects
        labeled, num_defects = ndimage.label(defects)
        
        defect_list = []
        for i in range(1, num_defects + 1):
            defect_mask = labeled == i
            size = np.sum(defect_mask)
            center = ndimage.center_of_mass(defect_mask)
            
            defect_list.append({
                'size': size * 5e-9,  # Convert to nm
                'position': center,
                'type': self.classify_defect(defect_mask)
            })
        
        return defect_list
    
    def monte_carlo_simulation(self, n_simulations=1000):
        """Monte Carlo process variation analysis"""
        cd_values = []
        overlay_errors = []
        
        for _ in range(n_simulations):
            # Random process variations
            dose_variation = np.random.normal(1.0, 0.05)
            focus_variation = np.random.normal(0, 20e-9)
            
            # Simulate with variations
            mask, _ = self.generate_mask_pattern()
            aerial = self.calculate_aerial_image(mask * dose_variation)
            
            # Add focus blur
            aerial = ndimage.gaussian_filter(aerial, sigma=abs(focus_variation)*1e9/10)
            
            # Measure CD
            cd = self.measure_cd(aerial)
            cd_values.append(cd)
            
            # Random overlay error
            overlay = np.random.normal(0, 5e-9)
            overlay_errors.append(overlay)
        
        return {
            'cd_mean': np.mean(cd_values),
            'cd_3sigma': 3 * np.std(cd_values),
            'overlay_mean': np.mean(overlay_errors),
            'overlay_3sigma': 3 * np.std(overlay_errors),
            'cpk': self.calculate_cpk(cd_values, target=45e-9, tolerance=5e-9)
        }

# Initialize and run simulation
sim = PhotolithographySimulator(wavelength=193e-9, NA=1.35, sigma=0.85)

# Generate mask with OPC
mask, phase = sim.generate_mask_pattern(pitch=65e-9)
target = mask.copy()
mask_opc = sim.apply_opc(mask, target, iterations=5)

# Calculate aerial image
aerial = sim.calculate_aerial_image(mask_opc)

# Detect defects
reference = aerial.copy()
reference[100:150, 100:150] = 0  # Introduce defect
defects = sim.detect_defects(reference, aerial)

# Monte Carlo analysis
mc_results = sim.monte_carlo_simulation(n_simulations=1000)

print(f"Resolution: {sim.calculate_resolution()*1e9:.1f} nm")
print(f"Defects found: {len(defects)}")
print(f"CD 3-sigma: {mc_results['cd_3sigma']*1e9:.2f} nm")
print(f"Process Cpk: {mc_results['cpk']:.2f}")
