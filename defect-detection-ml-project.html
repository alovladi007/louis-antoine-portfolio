Data Generation Pipeline

# Generate synthetic wafer maps with defects
from src.data_generator import DefectDataGenerator, DefectType

generator = DefectDataGenerator(wafer_size=300, die_size=(10, 10))

# Generate different defect patterns
wafer_map = generator.generate_wafer_map(
    defect_type=DefectType.CLUSTER,
    defect_density=0.05,
    seed=42
)

# Generate training dataset
dataset = generator.generate_dataset(
    n_samples=1000,
    defect_classes=["particle", "scratch", "void", "bridge", "none"],
    train_split=0.8
)

print(f"Yield Rate: {wafer_map.yield_rate:.1%}")
print(f"Defect Count: {wafer_map.defect_count}")

CNN Model Architecture

# Create and use different CNN architectures
from src.cnn_models import create_model
import torch

# Create ResNet-based classifier
model = create_model(
    model_name='resnet',
    num_classes=5,
    resnet_version='resnet50',
    pretrained=True
)

# Multi-scale CNN for various defect sizes
multiscale_model = create_model(
    model_name='multiscale_cnn',
    num_classes=5
)

# Attention-based CNN
attention_model = create_model(
    model_name='attention_cnn',
    num_classes=5
)

# Forward pass with attention
input_tensor = torch.randn(4, 1, 224, 224)
logits, attention_map = attention_model(input_tensor)

Bayesian Uncertainty Quantification

# MC Dropout for uncertainty estimation
from src.bayesian_models import MCDropoutClassifier

mc_model = MCDropoutClassifier(num_classes=5, dropout_rate=0.5)

# Predict with uncertainty
mean_pred, uncertainty = mc_model.predict_with_uncertainty(
    x=input_tensor,
    n_samples=50  # Monte Carlo samples
)

# Rare defect detection
from src.bayesian_models import RareDefectDetector

detector = RareDefectDetector(n_components=10, contamination=0.1)
detector.fit(X_normal_samples)

# Detect anomalies
predictions, anomaly_scores = detector.predict(X_test_samples)

Active Learning

# Select informative samples for labeling
from src.bayesian_models import ActiveLearningSelector

selector = ActiveLearningSelector(strategy='uncertainty')

# Select samples with highest uncertainty
selected_indices = selector.select_samples(
    model=mc_model,
    unlabeled_data=unlabeled_pool,
    n_samples=10

# Hybrid strategy combining uncertainty and diversity
hybrid_selector = ActiveLearningSelector(strategy='hybrid')
