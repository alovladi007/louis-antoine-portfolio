Mask Pattern Generation

# Generate various mask patterns
from src.mask_generation import MaskPatternGenerator, MaskParameters

# Configure parameters
params = MaskParameters(
    width=2048,
    height=2048,
    pixel_size=5.0,  # nm per pixel
    wavelength=193.0,  # ArF laser
    NA=0.75
)

generator = MaskPatternGenerator(params)

# Create line-space pattern
mask = generator.create_line_space_pattern(
    line_width=50,
    space_width=50,
    orientation='vertical'
)

# Add contact holes
mask = generator.create_contact_hole_array(
    hole_diameter=40,
    pitch=100
)

# Create resolution test pattern
test_pattern = generator.create_resolution_test_pattern()

# Add intentional defects for testing
generator.add_defect('particle', x=500, y=500, size=15)

# Export mask
generator.export_mask('mask.png', format='png')
generator.visualize_mask()

Optical Proximity Correction

# Apply OPC corrections
from src.opc import OpticalProximityCorrection, OPCParameters

opc_params = OPCParameters(
    wavelength=193,
    NA=0.75,
    k1=0.4,
    iterations=10
)

opc = OpticalProximityCorrection(opc_params)

# Apply rule-based OPC
corrected_mask = opc.apply_rule_based_opc(mask)

# Apply model-based OPC
optimized_mask = opc.apply_model_based_opc(
    mask,
    target_pattern=mask,
    pixel_size=5.0
)

# Evaluate OPC effectiveness
metrics = opc.evaluate_correction(pixel_size=5.0)
print(f"Edge Placement Error: {metrics['corrected_edge_placement_error']:.2f} nm")
print(f"Pattern Fidelity: {metrics['pattern_fidelity']:.2%}")

# Visualize results
opc.visualize_opc_results()

Fourier Optics Simulation

# Simulate optical imaging
from src.fourier_optics import FourierOpticsSimulator, OpticalSystem

# Define optical system
system = OpticalSystem(
    wavelength=193,
    NA=0.93,
    sigma_outer=0.7,
    sigma_inner=0.5,
    immersion=True,
    n_medium=1.44
)

simulator = FourierOpticsSimulator(system)

# Simulate different illumination modes
aerial_conventional = simulator.simulate_image_formation(
    mask, pixel_size=2.5, source_type='conventional'
)

aerial_annular = simulator.simulate_image_formation(
    mask, pixel_size=2.5, source_type='annular'
)

aerial_quadrupole = simulator.simulate_image_formation(
    mask, pixel_size=2.5, source_type='quadrupole'
)

# Calculate MTF
freq, mtf = simulator.calculate_mtf()

# Simulate focus-exposure matrix
focus_range = np.linspace(-100, 100, 21)  # nm

Monte Carlo Process Simulation

# Run Monte Carlo simulations
from src.monte_carlo import MonteCarloSimulator, ProcessParameters

params = ProcessParameters(
    exposure_dose=30.0,  # mJ/cm¬≤
    acid_diffusion_length=10.0,  # nm
    shot_noise=True,
    acid_diffusion=True,
    development_noise=True,
    n_photons=10000
)

mc_simulator = MonteCarloSimulator(params)

# Run full process simulation
results = mc_simulator.simulate_full_process(
    mask=mask,
    aerial_image=aerial_image,
    n_simulations=100
)

# Get statistics
print(f"LER Mean: {results['statistics']['line_edge_roughness']['mean']:.2f} nm")
print(f"CD Mean: {results['statistics']['critical_dimensions']['mean']:.2f} nm")
print(f"CD Std: {results['statistics']['critical_dimensions']['std']:.2f} nm")

# Parallel simulation for better performance
results_parallel = mc_simulator.parallel_simulate(
    mask, aerial_image, n_simulations=1000, n_processes=8
)

    MaskPatternGenerator,
    OpticalProximityCorrection,
    DefectInspector,
    FourierOpticsSimulator,
    MonteCarloSimulator
)

generator = MaskPatternGenerator()
mask = generator.create_line_space_pattern(50, 50)

opc = OpticalProximityCorrection()
corrected = opc.apply_rule_based_opc(mask)

simulator = FourierOpticsSimulator()
aerial = simulator.simulate_image_formation(corrected)
